\NewDocumentCommand{\ProcessGrounds}{mm}
 {
  \seq_set_split:Nnx \l_titles_seq { ; } { #1 }
  \seq_set_split:Nnx \l_sections_seq { || } { #2 }
  
  \int_set:Nn \l_vox_counter_int { 0 }
  \seq_map_inline:Nn \l_titles_seq {
    \int_incr:N \l_vox_counter_int
    \section*{\raggedright Ground~\int_to_arabic:n { \l_vox_counter_int }:~\tl_trim_spaces:n{##1}}
    \tl_set:Nx \l_current_reasons_tl { \seq_item:Nn \l_sections_seq { \l_vox_counter_int } }
    \seq_set_split:NnV \l_reasons_seq { ; } \l_current_reasons_tl
    
    \begin{enumerate}[label=\arabic*., start=\int_eval:n{\value{groundcounter}+1}, leftmargin=6ex]
      \seq_map_inline:Nn \l_reasons_seq {
        \stepcounter{groundcounter}
        \item \tl_trim_spaces:n { ####1 }
      }
    \end{enumerate}
  }
 }
